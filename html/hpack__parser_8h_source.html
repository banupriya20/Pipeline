40// Top level interface for parsing a sequence of header, continuation frames.
41class HPackParser {
42 public:
43 // What kind of stream boundary is provided by this frame?
44 enum class Boundary : uint8_t {
45 // More continuations are expected
46 None,
47 // This marks the end of headers, so data frames should follow
48 EndOfHeaders,
49 // This marks the end of headers *and* the end of the stream
50 EndOfStream
51 };
52 // What kind of priority is represented in the next frame
53 enum class Priority : uint8_t {
54 // No priority field
55 None,
56 // Yes there's a priority field
57 Included
58 };
59 // Details about a frame we only need to know for logging
60 struct LogInfo {
61 // The stream ID
62 uint32_t stream_id;
63 // Headers or trailers?
64 enum Type : uint8_t {
65 kHeaders,
66 kTrailers,
67 kDontKnow,
68 };
69 Type type;
70 // Client or server?
71 bool is_client;
72 };
73
74 HPackParser();
75 ~HPackParser();
76
77 // Non-copyable/movable
78 HPackParser(const HPackParser&) = delete;
79 HPackParser& operator=(const HPackParser&) = delete;
80
81 // Begin parsing a new frame
82 // Sink receives each parsed header,
83 void BeginFrame(grpc_metadata_batch* metadata_buffer,
84 uint32_t metadata_size_soft_limit,
85 uint32_t metadata_size_hard_limit, Boundary boundary,
86 Priority priority, LogInfo log_info);
87 // Start throwing away any received headers after parsing them.
88 void StopBufferingFrame() { metadata_buffer_ = nullptr; }
89 // Parse one slice worth of data
90 grpc_error_handle Parse(const grpc_slice& slice, bool is_last);
91 // Reset state ready for the next BeginFrame
92 void FinishFrame();
93
94 // Retrieve the associated hpack table (for tests, debugging)
95 HPackTable* hpack_table() { return &table_; }
96 // Is the current frame a boundary of some sort
97 bool is_boundary() const { return boundary_ != Boundary::None; }
98 // Is the current frame the end of a stream
99 bool is_eof() const { return boundary_ == Boundary::EndOfStream; }
100
101 private:
102 // Helper classes: see implementation
103 class Parser;
104 class Input;
105 class String;
106
107 grpc_error_handle ParseInput(Input input, bool is_last);
108 void ParseInputInner(Input* input);
109 GPR_ATTRIBUTE_NOINLINE
110 void HandleMetadataSoftSizeLimitExceeded(Input* input);
111
112 // Target metadata buffer
113 grpc_metadata_batch* metadata_buffer_ = nullptr;
114
115 // Bytes that could not be parsed last parsing round
116 std::vector<uint8_t> unparsed_bytes_;
117 // Buffer kind of boundary
118 // TODO(ctiller): see if we can move this argument to Parse, and avoid
119 // buffering.
120 Boundary boundary_;
121 // Buffer priority
122 // TODO(ctiller): see if we can move this argument to Parse, and avoid
123 // buffering.
124 Priority priority_;
125 uint8_t dynamic_table_updates_allowed_;
126 // Length of frame so far.
127 uint32_t frame_length_;
128 RandomEarlyDetection metadata_early_detection_;
129 // Information for logging
130 LogInfo log_info_;
131
132 // hpack table
133 HPackTable table_;
134};
