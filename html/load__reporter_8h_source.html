48// The interface to get the Census stats. Abstracted for mocking.
49class CensusViewProvider {
50 public:
51 // Maps from the view name to the view data.
52 using ViewDataMap =
53 std::unordered_map<std::string, ::opencensus::stats::ViewData>;
54 // Maps from the view name to the view descriptor.
55 using ViewDescriptorMap =
56 std::unordered_map<std::string, ::opencensus::stats::ViewDescriptor>;
57
58 CensusViewProvider();
59 virtual ~CensusViewProvider() = default;
60
61 // Fetches the view data accumulated since last fetching, and returns it as a
62 // map from the view name to the view data.
63 virtual ViewDataMap FetchViewData() = 0;
64
65 // A helper function that gets a row with the input tag values from the view
66 // data. Only used when we know that row must exist because we have seen a row
67 // with the same tag values in a related view data. Several ViewData's are
68 // considered related if their views are based on the measures that are always
69 // recorded at the same time.
70 static double GetRelatedViewDataRowDouble(
71 const ViewDataMap& view_data_map, const char* view_name,
72 size_t view_name_len, const std::vector<std::string>& tag_values);
73 static uint64_t GetRelatedViewDataRowInt(
74 const ViewDataMap& view_data_map, const char* view_name,
75 size_t view_name_len, const std::vector<std::string>& tag_values);
76
77 protected:
78 const ViewDescriptorMap& view_descriptor_map() const {
79 return view_descriptor_map_;
80 }
81
82 private:
83 ViewDescriptorMap view_descriptor_map_;
84 // Tag keys.
85 ::opencensus::tags::TagKey tag_key_token_;
86 ::opencensus::tags::TagKey tag_key_host_;
87 ::opencensus::tags::TagKey tag_key_user_id_;
88 ::opencensus::tags::TagKey tag_key_status_;
89 ::opencensus::tags::TagKey tag_key_metric_name_;
90};

122// Maintains all the load data and load reporting streams.
123class LoadReporter {
124 public:
125 // TODO(juanlishen): Allow config for providers from users.
126 LoadReporter(uint32_t feedback_sample_window_seconds,
127 std::unique_ptr<CensusViewProvider> census_view_provider,
128 std::unique_ptr<CpuStatsProvider> cpu_stats_provider)
129 : feedback_sample_window_seconds_(feedback_sample_window_seconds),
130 census_view_provider_(std::move(census_view_provider)),
131 cpu_stats_provider_(std::move(cpu_stats_provider)) {
132 // Append the initial record so that the next real record can have a base.
133 AppendNewFeedbackRecord(0, 0);
134 }
135
136 // Fetches the latest data from Census and merge it into the data store.
137 // Also adds a new sample to the LB feedback sliding window.
138 // Thread-unsafe. (1). The access to the load data store and feedback records
139 // has locking. (2). The access to the Census view provider and CPU stats
140 // provider lacks locking, but we only access these two members in this method
141 // (in testing, we also access them when setting up expectation). So the
142 // invocations of this method must be serialized.
143 void FetchAndSample();
144
145 // Generates a report for that host and balancer. The report contains
146 // all the stats data accumulated between the last report (i.e., the last
147 // consumption) and the last fetch from Census (i.e., the last production).
148 // Thread-safe.
149 ::google::protobuf::RepeatedPtrField<grpc::lb::v1::Load> GenerateLoads(
150 const std::string& hostname, const std::string& lb_id);
151
152 // The feedback is calculated from the stats data recorded in the sliding
153 // window. Outdated records are discarded.
154 // Thread-safe.
155 grpc::lb::v1::LoadBalancingFeedback GenerateLoadBalancingFeedback();
156
157 // Wrapper around LoadDataStore::ReportStreamCreated.
158 // Thread-safe.
159 void ReportStreamCreated(const std::string& hostname,
160 const std::string& lb_id,
161 const std::string& load_key);
162
163 // Wrapper around LoadDataStore::ReportStreamClosed.
164 // Thread-safe.
165 void ReportStreamClosed(const std::string& hostname,
166 const std::string& lb_id);
167
168 // Generates a unique LB ID of length kLbIdLength. Returns an empty string
169 // upon failure. Thread-safe.
170 std::string GenerateLbId();
171
172 // Accessors only for testing.
173 CensusViewProvider* census_view_provider() {
174 return census_view_provider_.get();
175 }
176 CpuStatsProvider* cpu_stats_provider() { return cpu_stats_provider_.get(); }
177
178 private:
179 struct LoadBalancingFeedbackRecord {
180 std::chrono::system_clock::time_point end_time;
181 uint64_t rpcs;
182 uint64_t errors;
183 uint64_t cpu_usage;
184 uint64_t cpu_limit;
185
186 LoadBalancingFeedbackRecord(
187 const std::chrono::system_clock::time_point& end_time, uint64_t rpcs,
188 uint64_t errors, uint64_t cpu_usage, uint64_t cpu_limit)
189 : end_time(end_time),
190 rpcs(rpcs),
191 errors(errors),
192 cpu_usage(cpu_usage),
193 cpu_limit(cpu_limit) {}
194 };

196 // Finds the view data about starting call from the view_data_map and merges
197 // the data to the load data store.
198 void ProcessViewDataCallStart(
199 const CensusViewProvider::ViewDataMap& view_data_map);
200 // Finds the view data about ending call from the view_data_map and merges the
201 // data to the load data store.
202 void ProcessViewDataCallEnd(
203 const CensusViewProvider::ViewDataMap& view_data_map);
204 // Finds the view data about the customized call metrics from the
205 // view_data_map and merges the data to the load data store.
206 void ProcessViewDataOtherCallMetrics(
207 const CensusViewProvider::ViewDataMap& view_data_map);
208
209 bool IsRecordInWindow(const LoadBalancingFeedbackRecord& record,
210 std::chrono::system_clock::time_point now) {
211 return record.end_time > now - feedback_sample_window_seconds_;
212 }
213
214 void AppendNewFeedbackRecord(uint64_t rpcs, uint64_t errors);
215
216 // Extracts an OrphanedLoadIdentifier from the per-balancer store and attaches
217 // it to the load.
218 void AttachOrphanLoadId(grpc::lb::v1::Load* load,
219 const PerBalancerStore& per_balancer_store);
220
221 std::atomic<int64_t> next_lb_id_{0};
222 const std::chrono::seconds feedback_sample_window_seconds_;
223 grpc_core::Mutex feedback_mu_;
224 std::deque<LoadBalancingFeedbackRecord> feedback_records_;
225 // TODO(juanlishen): Lock in finer grain. Locking the whole store may be
226 // too expensive.
227 grpc_core::Mutex store_mu_;
228 LoadDataStore load_data_store_;
229 std::unique_ptr<CensusViewProvider> census_view_provider_;
230 std::unique_ptr<CpuStatsProvider> cpu_stats_provider_;
231};
