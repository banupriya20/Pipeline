32// Tracks errors that occur during validation of a data structure (e.g.,
33// a JSON object or protobuf message). Errors are tracked based on
34// which field they are associated with. If at least one error occurs
35// during validation, the validation failed.
36//
37// Example usage:
38//
39// absl::StatusOr<std::string> GetFooBar(const Json::Object& json) {
40// ValidationErrors errors;
41// {
42// ValidationErrors::ScopedField field("foo");
43// auto it = json.object().find("foo");
44// if (it == json.object().end()) {
45// errors.AddError("field not present");
46// } else if (it->second.type() != Json::Type::kObject) {
47// errors.AddError("must be a JSON object");
48// } else {
49// const Json& foo = it->second;
50// ValidationErrors::ScopedField field(".bar");
51// auto it = foo.object().find("bar");
52// if (it == json.object().end()) {
53// errors.AddError("field not present");
54// } else if (it->second.type() != Json::Type::kString) {
55// errors.AddError("must be a JSON string");
56// } else {
57// return it->second.string();
58// }
59// }
60// }
61// return errors.status(absl::StatusCode::kInvalidArgument,
62// "errors validating foo.bar");
63// }
64class ValidationErrors {
65 public:
66 // Pushes a field name onto the stack at construction and pops it off
67 // of the stack at destruction.
68 class ScopedField {
69 public:
70 ScopedField(ValidationErrors* errors, absl::string_view field_name)
71 : errors_(errors) {
72 errors_->PushField(field_name);
73 }
74
75 // Not copyable.
76 ScopedField(const ScopedField& other) = delete;
77 ScopedField& operator=(const ScopedField& other) = delete;
78
79 // Movable.
80 ScopedField(ScopedField&& other) noexcept
81 : errors_(std::exchange(other.errors_, nullptr)) {}
82 ScopedField& operator=(ScopedField&& other) noexcept {
83 if (errors_ != nullptr) errors_->PopField();
84 errors_ = std::exchange(other.errors_, nullptr);
85 return *this;
86 }
87
88 ~ScopedField() {
89 if (errors_ != nullptr) errors_->PopField();
90 }
91
92 private:
93 ValidationErrors* errors_;
94 };
95
96 // Records that we've encountered an error associated with the current
97 // field.
98 void AddError(absl::string_view error) GPR_ATTRIBUTE_NOINLINE;
99
100 // Returns true if the current field has errors.
101 bool FieldHasErrors() const GPR_ATTRIBUTE_NOINLINE;
102
103 // Returns the resulting status of parsing.
104 // If there are no errors, this will return an Ok status instead of using the
105 // prefix argument.
106 absl::Status status(absl::StatusCode code, absl::string_view prefix) const;
107
108 // Returns the resulting error message
109 // If there are no errors, this will return an empty string.
110 std::string message(absl::string_view prefix) const;
111
112 // Returns true if there are no errors.
113 bool ok() const { return field_errors_.empty(); }
114
115 size_t size() const { return field_errors_.size(); }
116
117 private:
118 // Pushes a field name onto the stack.
119 void PushField(absl::string_view ext) GPR_ATTRIBUTE_NOINLINE;
120 // Pops a field name off of the stack.
121 void PopField() GPR_ATTRIBUTE_NOINLINE;
122
123 // Errors that we have encountered so far, keyed by field name.
124 // TODO(roth): If we don't actually have any fields for which we
125 // report more than one error, simplify this data structure.
126 std::map<std::string /*field_name*/, std::vector<std::string>> field_errors_;
127 // Stack of field names indicating the field that we are currently
128 // validating.
129 std::vector<std::string> fields_;
130};
