150// Move only type that tracks call startup.
151// Allows observation of when client_initial_metadata has been processed by the
152// end of the local call stack.
153// Interested observers can call Wait() to obtain a promise that will resolve
154// when all local client_initial_metadata processing has completed.
155// The result of this token is either true on successful completion, or false
156// if the metadata was not sent.
157// To set a successful completion, call Complete(true). For failure, call
158// Complete(false).
159// If Complete is not called, the destructor of a still held token will complete
160// with failure.
161// Transports should hold this token until client_initial_metadata has passed
162// any flow control (eg MAX_CONCURRENT_STREAMS for http2).
163class ClientInitialMetadataOutstandingToken {
164 public:
165 static ClientInitialMetadataOutstandingToken Empty() {
166 return ClientInitialMetadataOutstandingToken();
167 }
168 static ClientInitialMetadataOutstandingToken New(
169 Arena* arena = GetContext<Arena>()) {
170 ClientInitialMetadataOutstandingToken token;
171 token.latch_ = arena->New<Latch<bool>>();
172 return token;
173 }
174
175 ClientInitialMetadataOutstandingToken(
176 const ClientInitialMetadataOutstandingToken&) = delete;
177 ClientInitialMetadataOutstandingToken& operator=(
178 const ClientInitialMetadataOutstandingToken&) = delete;
179 ClientInitialMetadataOutstandingToken(
180 ClientInitialMetadataOutstandingToken&& other) noexcept
181 : latch_(std::exchange(other.latch_, nullptr)) {}
182 ClientInitialMetadataOutstandingToken& operator=(
183 ClientInitialMetadataOutstandingToken&& other) noexcept {
184 latch_ = std::exchange(other.latch_, nullptr);
185 return *this;
186 }
187 ~ClientInitialMetadataOutstandingToken() {
188 if (latch_ != nullptr) latch_->Set(false);
189 }
190 void Complete(bool success) { std::exchange(latch_, nullptr)->Set(success); }
191
192 // Returns a promise that will resolve when this object (or its moved-from
193 // ancestor) is dropped.
194 auto Wait() { return latch_->Wait(); }
195
196 private:
197 ClientInitialMetadataOutstandingToken() = default;
198
199 Latch<bool>* latch_ = nullptr;
200};
201
202using ClientInitialMetadataOutstandingTokenWaitType =
203 decltype(std::declval<ClientInitialMetadataOutstandingToken>().Wait());

205struct CallArgs {
206 // Initial metadata from the client to the server.
207 // During promise setup this can be manipulated by filters (and then
208 // passed on to the next filter).
209 ClientMetadataHandle client_initial_metadata;
210 // Token indicating that client_initial_metadata is still being processed.
211 // This should be moved around and only destroyed when the transport is
212 // satisfied that the metadata has passed any flow control measures it has.
213 ClientInitialMetadataOutstandingToken client_initial_metadata_outstanding;
214 // Latch that will ultimately contain the polling entity for the call.
215 // TODO(ctiller): remove once event engine lands
216 Latch<grpc_polling_entity>* polling_entity;
217 // Initial metadata from the server to the client.
218 // Set once when it's available.
219 // During promise setup filters can substitute their own latch for this
220 // and consequently intercept the sent value and mutate/observe it.
221 PipeSender<ServerMetadataHandle>* server_initial_metadata;
222 // Messages travelling from the application to the transport.
223 PipeReceiver<MessageHandle>* client_to_server_messages;
224 // Messages travelling from the transport to the application.
225 PipeSender<MessageHandle>* server_to_client_messages;
226};

325// This struct (which is present in both grpc_transport_stream_op_batch
326// and grpc_transport_op_batch) is a convenience to allow filters or
327// transports to schedule a closure related to a particular batch without
328// having to allocate memory. The general pattern is to initialize the
329// closure with the callback arg set to the batch and extra_arg set to
330// whatever state is associated with the handler (e.g., the call element
331// or the transport stream object).
332//
333// Note that this can only be used by the current handler of a given
334// batch on the way down the stack (i.e., whichever filter or transport is
335// currently handling the batch). Once a filter or transport passes control
336// of the batch to the next handler, it cannot depend on the contents of
337// this struct anymore, because the next handler may reuse it.
338struct grpc_handler_private_op_data {
339 void* extra_arg = nullptr;
340 grpc_closure closure;
341 grpc_handler_private_op_data() { memset(&closure, 0, sizeof(closure)); }
342};

412struct grpc_transport_stream_op_batch_payload {
413 explicit grpc_transport_stream_op_batch_payload(
414 grpc_call_context_element* context)
415 : context(context) {}
416 struct {
417 grpc_metadata_batch* send_initial_metadata = nullptr;
418 } send_initial_metadata;
419
420 struct {
421 grpc_metadata_batch* send_trailing_metadata = nullptr;
422 // Set by the transport to true if the stream successfully wrote the
423 // trailing metadata. If this is not set but there was a send trailing
424 // metadata op present, this can indicate that a server call can be marked
425 // as a cancellation (since the stream was write-closed before status could
426 // be delivered).
427 bool* sent = nullptr;
428 } send_trailing_metadata;
429
430 struct {
431 // The transport (or a filter that decides to return a failure before
432 // the op gets down to the transport) takes ownership.
433 // The batch's on_complete will not be called until after the byte
434 // stream is orphaned.
435 grpc_core::SliceBuffer* send_message;
436 uint32_t flags = 0;
437 // Set by the transport if the stream has been closed for writes. If this
438 // is set and send message op is present, we set the operation to be a
439 // failure without sending a cancel OP down the stack. This is so that the
440 // status of the call does not get overwritten by the Cancel OP, which would
441 // be especially problematic if we had received a valid status from the
442 // server.
443 // For send_initial_metadata, it is fine for the status to be overwritten
444 // because at that point, the client will not have received a status.
445 // For send_trailing_metadata, we might overwrite the status if we have
446 // non-zero metadata to send. This is fine because the API does not allow
447 // the client to send trailing metadata.
448 bool stream_write_closed = false;
449 } send_message;

459 // uses this to set the success flag of OnReadInitialMetadataDone()
