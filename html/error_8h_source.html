57// Consumes all the errors in the vector and forms a referencing error from
58// them. If the vector is empty, return absl::OkStatus().
59template <typename VectorType>
60static absl::Status grpc_status_create_from_vector(
61 const grpc_core::DebugLocation& location, absl::string_view desc,
62 VectorType* error_list) {
63 absl::Status error;
64 if (error_list->size() != 0) {
65 error = grpc_status_create(absl::StatusCode::kUnknown, desc, location,
66 error_list->size(), error_list->data());
67 error_list->clear();
68 }
69 return error;
70}

134class AtomicError {
135 public:
136 AtomicError() = default;
137 explicit AtomicError(grpc_error_handle error) { error_ = error; }
138
139 AtomicError(const AtomicError&) = delete;
140 AtomicError& operator=(const AtomicError&) = delete;
141
143 bool ok() {
144 gpr_spinlock_lock(&lock_);
145 bool ret = error_.ok();
146 gpr_spinlock_unlock(&lock_);
147 return ret;
148 }
149
150 grpc_error_handle get() {
151 gpr_spinlock_lock(&lock_);
152 grpc_error_handle ret = error_;
153 gpr_spinlock_unlock(&lock_);
154 return ret;
155 }
156
157 void set(grpc_error_handle error) {
158 gpr_spinlock_lock(&lock_);
159 error_ = error;
160 gpr_spinlock_unlock(&lock_);
161 }
162
163 private:
164 grpc_error_handle error_;
165 gpr_spinlock lock_ = GPR_SPINLOCK_STATIC_INITIALIZER;
166};
