49// The base class for all tracer implementations.
50class CallTracerAnnotationInterface {
51 public:
52 virtual ~CallTracerAnnotationInterface() {}
53 // Records an annotation on the call attempt.
54 // TODO(yashykt): If needed, extend this to attach attributes with
55 // annotations.
56 virtual void RecordAnnotation(absl::string_view annotation) = 0;
57 virtual std::string TraceId() = 0;
58 virtual std::string SpanId() = 0;
59 virtual bool IsSampled() = 0;
60};

62// The base class for CallAttemptTracer and ServerCallTracer.
63// TODO(yashykt): What's a better name for this?
64class CallTracerInterface : public CallTracerAnnotationInterface {
65 public:
66 ~CallTracerInterface() override {}
67 // Please refer to `grpc_transport_stream_op_batch_payload` for details on
68 // arguments.
69 virtual void RecordSendInitialMetadata(
70 grpc_metadata_batch* send_initial_metadata) = 0;
71 virtual void RecordSendTrailingMetadata(
72 grpc_metadata_batch* send_trailing_metadata) = 0;
73 virtual void RecordSendMessage(const SliceBuffer& send_message) = 0;
74 // Only invoked if message was actually compressed.
75 virtual void RecordSendCompressedMessage(
76 const SliceBuffer& send_compressed_message) = 0;
77 // The `RecordReceivedInitialMetadata()` and `RecordReceivedMessage()`
78 // methods should only be invoked when the metadata/message was
79 // successfully received, i.e., without any error.
80 virtual void RecordReceivedInitialMetadata(
81 grpc_metadata_batch* recv_initial_metadata) = 0;
82 virtual void RecordReceivedMessage(const SliceBuffer& recv_message) = 0;
83 // Only invoked if message was actually decompressed.
84 virtual void RecordReceivedDecompressedMessage(
85 const SliceBuffer& recv_decompressed_message) = 0;
86 virtual void RecordCancel(grpc_error_handle cancel_error) = 0;
87};

89// Interface for a tracer that records activities on a call. Actual attempts for
90// this call are traced with CallAttemptTracer after invoking RecordNewAttempt()
91// on the ClientCallTracer object.
92class ClientCallTracer : public CallTracerAnnotationInterface {
93 public:
94 // Interface for a tracer that records activities on a particular call
95 // attempt.
96 // (A single RPC can have multiple attempts due to retry/hedging policies or
97 // as transparent retry attempts.)
98 class CallAttemptTracer : public CallTracerInterface {
99 public:
100 ~CallAttemptTracer() override {}
101 // TODO(yashykt): The following two methods `RecordReceivedTrailingMetadata`
102 // and `RecordEnd` should be moved into CallTracerInterface.
103 // If the call was cancelled before the recv_trailing_metadata op
104 // was started, recv_trailing_metadata and transport_stream_stats
105 // will be null.
106 virtual void RecordReceivedTrailingMetadata(
107 absl::Status status, grpc_metadata_batch* recv_trailing_metadata,
108 const grpc_transport_stream_stats* transport_stream_stats) = 0;
109 // Should be the last API call to the object. Once invoked, the tracer
110 // library is free to destroy the object.
111 virtual void RecordEnd(const gpr_timespec& latency) = 0;
112 };
113
114 ~ClientCallTracer() override {}
115
116 // Records a new attempt for the associated call. \a transparent denotes
117 // whether the attempt is being made as a transparent retry or as a
118 // non-transparent retry/heding attempt. (There will be at least one attempt
119 // even if the call is not being retried.) The `ClientCallTracer` object
120 // retains ownership to the newly created `CallAttemptTracer` object.
121 // RecordEnd() serves as an indication that the call stack is done with all
122 // API calls, and the tracer library is free to destroy it after that.
123 virtual CallAttemptTracer* StartNewAttempt(bool is_transparent_retry) = 0;
124};

126// Interface for a tracer that records activities on a server call.
127class ServerCallTracer : public CallTracerInterface {
128 public:
129 ~ServerCallTracer() override {}
130 // TODO(yashykt): The following two methods `RecordReceivedTrailingMetadata`
131 // and `RecordEnd` should be moved into CallTracerInterface.
132 virtual void RecordReceivedTrailingMetadata(
133 grpc_metadata_batch* recv_trailing_metadata) = 0;
134 // Should be the last API call to the object. Once invoked, the tracer
135 // library is free to destroy the object.
136 virtual void RecordEnd(const grpc_call_final_info* final_info) = 0;
137};
