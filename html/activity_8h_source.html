51// A Wakeable object is used by queues to wake activities.
52class Wakeable {
53 public:
54 // Wake up the underlying activity.
55 // After calling, this Wakeable cannot be used again.
56 // WakeupMask comes from the activity that created this Wakeable and specifies
57 // the set of promises that should be awoken.
58 virtual void Wakeup(WakeupMask wakeup_mask) = 0;
59 // Drop this wakeable without waking up the underlying activity.
60 virtual void Drop(WakeupMask wakeup_mask) = 0;
61
62 // Return the underlying activity debug tag, or "<unknown>" if not available.
63 virtual std::string ActivityDebugTag(WakeupMask wakeup_mask) const = 0;
64
65 protected:
66 inline ~Wakeable() {}
67};

80// An owning reference to a Wakeable.
81// This type is non-copyable but movable.
82class Waker {
83 public:
84 Waker(Wakeable* wakeable, WakeupMask wakeup_mask)
85 : wakeable_and_arg_{wakeable, wakeup_mask} {}
86 Waker() : Waker(promise_detail::unwakeable(), 0) {}
87 ~Waker() { wakeable_and_arg_.Drop(); }
88 Waker(const Waker&) = delete;
89 Waker& operator=(const Waker&) = delete;
90 Waker(Waker&& other) noexcept : wakeable_and_arg_(other.Take()) {}
91 Waker& operator=(Waker&& other) noexcept {
92 std::swap(wakeable_and_arg_, other.wakeable_and_arg_);
93 return *this;
94 }
95
96 // Wake the underlying activity.
97 void Wakeup() { Take().Wakeup(); }
98
99 template <typename H>
100 friend H AbslHashValue(H h, const Waker& w) {
101 return H::combine(H::combine(std::move(h), w.wakeable_and_arg_.wakeable),
102 w.wakeable_and_arg_.wakeup_mask);
103 }
104
105 bool operator==(const Waker& other) const noexcept {
106 return wakeable_and_arg_ == other.wakeable_and_arg_;
107 }
108
109 bool operator!=(const Waker& other) const noexcept {
110 return !operator==(other);
111 }
112
113 std::string ActivityDebugTag() {
114 return wakeable_and_arg_.ActivityDebugTag();
115 }
116
117 // This is for tests to assert that a waker is occupied or not.
118 bool is_unwakeable() const {
119 return wakeable_and_arg_.wakeable == promise_detail::unwakeable();
120 }
121
122 private:
123 struct WakeableAndArg {
124 Wakeable* wakeable;
125 WakeupMask wakeup_mask;
126
127 void Wakeup() { wakeable->Wakeup(wakeup_mask); }
128 void Drop() { wakeable->Drop(wakeup_mask); }
129 std::string ActivityDebugTag() const {
130 return wakeable == nullptr ? "<unknown>"
131 : wakeable->ActivityDebugTag(wakeup_mask);
132 }
133 bool operator==(const WakeableAndArg& other) const noexcept {
134 return wakeable == other.wakeable && wakeup_mask == other.wakeup_mask;
135 }
136 };
137
138 WakeableAndArg Take() {
139 return std::exchange(wakeable_and_arg_, {promise_detail::unwakeable(), 0});
140 }
141
142 WakeableAndArg wakeable_and_arg_;
143};

161// An Activity tracks execution of a single promise.
162// It executes the promise under a mutex.
163// When the promise stalls, it registers the containing activity to be woken up
164// later.
165// The activity takes a callback, which will be called exactly once with the
166// result of execution.
167// Activity execution may be cancelled by simply deleting the activity. In such
168// a case, if execution had not already finished, the done callback would be
169// called with absl::CancelledError().
170class Activity : public Orphanable {
171 public:
172 // Force wakeup from the outside.
173 // This should be rarely needed, and usages should be accompanied with a note
174 // on why it's not possible to wakeup with a Waker object.
175 // Nevertheless, it's sometimes useful for integrations with Activity to force
176 // an Activity to repoll.
177 void ForceWakeup() { MakeOwningWaker().Wakeup(); }
178
179 // Force the current activity to immediately repoll if it doesn't complete.
180 virtual void ForceImmediateRepoll(WakeupMask mask) = 0;
181 // Legacy version of ForceImmediateRepoll() that uses the current participant.
182 // Will go away once Party gets merged with Activity. New usage is banned.
183 void ForceImmediateRepoll() { ForceImmediateRepoll(CurrentParticipant()); }
184
185 // Return the current part of the activity as a bitmask
186 virtual WakeupMask CurrentParticipant() const { return 1; }
187
188 // Return the current activity.
189 // Additionally:
190 // - assert that there is a current activity (and catch bugs if there's not)
191 // - indicate to thread safety analysis that the current activity is indeed
192 // locked
193 // - back up that assertation with a runtime check in debug builds (it's
194 // prohibitively expensive in non-debug builds)
195 static Activity* current() { return g_current_activity_; }
196
197 // Produce an activity-owning Waker. The produced waker will keep the activity
198 // alive until it's awoken or dropped.
199 virtual Waker MakeOwningWaker() = 0;
200
201 // Produce a non-owning Waker. The waker will own a small heap allocated weak
202 // pointer to this activity. This is more suitable for wakeups that may not be
203 // delivered until long after the activity should be destroyed.
204 virtual Waker MakeNonOwningWaker() = 0;
205
206 // Some descriptive text to add to log messages to identify this activity.
207 virtual std::string DebugTag() const;
208
209 protected:
210 // Check if this activity is the current activity executing on the current
211 // thread.
212 bool is_current() const { return this == g_current_activity_; }
213 // Check if there is an activity executing on the current thread.
214 static bool have_current() { return g_current_activity_ != nullptr; }
215 // Set the current activity at construction, clean it up at destruction.
216 class ScopedActivity {
217 public:
218 explicit ScopedActivity(Activity* activity)
219 : prior_activity_(g_current_activity_) {
220 g_current_activity_ = activity;
221 }
222 ~ScopedActivity() { g_current_activity_ = prior_activity_; }
223 ScopedActivity(const ScopedActivity&) = delete;
224 ScopedActivity& operator=(const ScopedActivity&) = delete;
225
226 private:
227 Activity* const prior_activity_;
228 };
229
230 private:
231 // Set during RunLoop to the Activity that's executing.
232 // Being set implies that mu_ is held.
233 static thread_local Activity* g_current_activity_;
234};

299// A free standing activity: an activity that owns its own synchronization and
300// memory.
301// The alternative is an activity that's somehow tied into another system, for
302// instance the type seen in promise_based_filter.h as we're transitioning from
303// the old filter stack to the new system.
304// FreestandingActivity is-a Wakeable, but needs to increment a refcount before
305// returning that Wakeable interface. Additionally, we want to keep
306// FreestandingActivity as small as is possible, since it will be used
307// everywhere. So we use inheritance to provide the Wakeable interface: this
308// makes it zero sized, and we make the inheritance private to prevent
309// accidental casting.
310class FreestandingActivity : public Activity, private Wakeable {
311 public:
312 Waker MakeOwningWaker() final {
313 Ref();
314 return Waker(this, 0);
315 }
316 Waker MakeNonOwningWaker() final;
317
318 void Orphan() final {
319 Cancel();
320 Unref();
321 }
322
323 void ForceImmediateRepoll(WakeupMask) final {
324 mu_.AssertHeld();
325 SetActionDuringRun(ActionDuringRun::kWakeup);
326 }
327
328 protected:
329 // Action received during a run, in priority order.
330 // If more than one action is received during a run, we use max() to resolve
331 // which one to report (so Cancel overrides Wakeup).
332 enum class ActionDuringRun : uint8_t {
333 kNone, // No action occured during run.
334 kWakeup, // A wakeup occured during run.
335 kCancel, // Cancel was called during run.
336 };
337
338 inline ~FreestandingActivity() override {
339 if (handle_) {
340 DropHandle();
341 }
342 }
343
344 // Check if we got an internal wakeup since the last time this function was
345 // called.
346 ActionDuringRun GotActionDuringRun() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_) {
347 return std::exchange(action_during_run_, ActionDuringRun::kNone);
348 }
349
350 // Implementors of Wakeable::Wakeup should call this after the wakeup has
351 // completed.
352 void WakeupComplete() { Unref(); }
353
354 // Set the action that occured during this run.
355 // We use max to combine actions so that cancellation overrides wakeups.
356 void SetActionDuringRun(ActionDuringRun action)
357 ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_) {
358 action_during_run_ = std::max(action_during_run_, action);
359 }
360
361 Mutex* mu() ABSL_LOCK_RETURNED(mu_) { return &mu_; }
362
363 std::string ActivityDebugTag(WakeupMask) const override { return DebugTag(); }
364
365 private:
366 class Handle;
367
368 // Cancel execution of the underlying promise.
369 virtual void Cancel() = 0;
370
371 void Ref() { refs_.fetch_add(1, std::memory_order_relaxed); }
372 void Unref() {
373 if (1 == refs_.fetch_sub(1, std::memory_order_acq_rel)) {
374 delete this;
375 }
376 }
377
378 // Return a Handle instance with a ref so that it can be stored waiting for
379 // some wakeup.
380 Handle* RefHandle() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
381 // If our refcount is non-zero, ref and return true.
382 // Otherwise, return false.
383 bool RefIfNonzero();
384 // Drop the (proved existing) wait handle.
385 void DropHandle() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
386
387 // All promise execution occurs under this mutex.
388 Mutex mu_;
389
390 // Current refcount.
391 std::atomic<uint32_t> refs_{1};
392 // If wakeup is called during Promise polling, we set this to Wakeup and
393 // repoll. If cancel is called during Promise polling, we set this to Cancel
394 // and cancel at the end of polling.
395 ActionDuringRun action_during_run_ ABSL_GUARDED_BY(mu_) =
396 ActionDuringRun::kNone;
397 // Handle for long waits. Allows a very small weak pointer type object to
398 // queue for wakeups while Activity may be deleted earlier.
399 Handle* handle_ ABSL_GUARDED_BY(mu_) = nullptr;
400};

402// Implementation details for an Activity of an arbitrary type of promise.
403// There should exist an inner template class `BoundScheduler` that provides
404// the following interface:
405// struct WakeupScheduler {
406// template <typename ActivityType>
407// class BoundScheduler {
408// public:
409// BoundScheduler(WakeupScheduler);
410// void ScheduleWakeup();
411// };
412// };
413// The ScheduleWakeup function should arrange that
414// static_cast<ActivityType*>(this)->RunScheduledWakeup() be invoked at the
415// earliest opportunity.
416// It can assume that activity will remain live until RunScheduledWakeup() is
417// invoked, and that a given activity will not be concurrently scheduled again
418// until its RunScheduledWakeup() has been invoked.
419// We use private inheritance here as a way of getting private members for each
420// of the contexts.
421// TODO(ctiller): We can probably reconsider the private inheritance here
422// when we move away from C++11 and have more powerful template features.
423template <class F, class WakeupScheduler, class OnDone, typename... Contexts>
424class PromiseActivity final
425 : public FreestandingActivity,
426 public WakeupScheduler::template BoundScheduler<
427 PromiseActivity<F, WakeupScheduler, OnDone, Contexts...>>,
428 private ActivityContexts<Contexts...> {
429 public:
430 using Factory = OncePromiseFactory<void, F>;
431 using ResultType = typename Factory::Promise::Result;
432
433 PromiseActivity(F promise_factory, WakeupScheduler wakeup_scheduler,
434 OnDone on_done, Contexts&&... contexts)
435 : FreestandingActivity(),
436 WakeupScheduler::template BoundScheduler<PromiseActivity>(
437 std::move(wakeup_scheduler)),
438 ActivityContexts<Contexts...>(std::forward<Contexts>(contexts)...),
439 on_done_(std::move(on_done)) {
440 // Lock, construct an initial promise from the factory, and step it.
441 // This may hit a waiter, which could expose our this pointer to other
442 // threads, meaning we do need to hold this mutex even though we're still
443 // constructing.
444 mu()->Lock();
445 auto status = Start(Factory(std::move(promise_factory)));
446 mu()->Unlock();
447 // We may complete immediately.
448 if (status.has_value()) {
449 on_done_(std::move(*status));
450 }
451 }
452
453 ~PromiseActivity() override {
454 // We shouldn't destruct without calling Cancel() first, and that must get
455 // us to be done_, so we assume that and have no logic to destruct the
456 // promise here.
457 GPR_ASSERT(done_);
458 }
459
460 void RunScheduledWakeup() {
461 GPR_ASSERT(wakeup_scheduled_.exchange(false, std::memory_order_acq_rel));
462 Step();
463 WakeupComplete();
464 }
465
466 private:
467 using typename ActivityContexts<Contexts...>::ScopedContext;
468
469 void Cancel() final {
470 if (Activity::is_current()) {
471 mu()->AssertHeld();
472 SetActionDuringRun(ActionDuringRun::kCancel);
473 return;
474 }
475 bool was_done;
476 {
477 MutexLock lock(mu());
478 // Check if we were done, and flag done.
479 was_done = done_;
480 if (!done_) {
481 ScopedActivity scoped_activity(this);
482 ScopedContext contexts(this);
483 MarkDone();
484 }
485 }
486 // If we were not done, then call the on_done callback.
487 if (!was_done) {
488 on_done_(absl::CancelledError());
489 }
490 }
491
492 // Wakeup this activity. Arrange to poll the activity again at a convenient
493 // time: this could be inline if it's deemed safe, or it could be by passing
494 // the activity to an external threadpool to run. If the activity is already
495 // running on this thread, a note is taken of such and the activity is
496 // repolled if it doesn't complete.
497 void Wakeup(WakeupMask) final {
498 // If there is an active activity, but hey it's us, flag that and we'll loop
499 // in RunLoop (that's calling from above here!).
500 if (Activity::is_current()) {
501 mu()->AssertHeld();
502 SetActionDuringRun(ActionDuringRun::kWakeup);
503 WakeupComplete();
504 return;
505 }
506 if (!wakeup_scheduled_.exchange(true, std::memory_order_acq_rel)) {
507 // Can't safely run, so ask to run later.
508 this->ScheduleWakeup();
509 } else {
510 // Already a wakeup scheduled for later, drop ref.
511 WakeupComplete();
512 }
513 }
514
515 // Drop a wakeup
516 void Drop(WakeupMask) final { this->WakeupComplete(); }
517
518 // Notification that we're no longer executing - it's ok to destruct the
519 // promise.
520 void MarkDone() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu()) {
521 GPR_ASSERT(!std::exchange(done_, true));
522 ScopedContext contexts(this);
523 Destruct(&promise_holder_.promise);
524 }
525
526 // In response to Wakeup, run the Promise state machine again until it
527 // settles. Then check for completion, and if we have completed, call on_done.
528 void Step() ABSL_LOCKS_EXCLUDED(mu()) {
529 // Poll the promise until things settle out under a lock.
530 mu()->Lock();
531 if (done_) {
532 // We might get some spurious wakeups after finishing.
533 mu()->Unlock();
534 return;
535 }
536 auto status = RunStep();
537 mu()->Unlock();
538 if (status.has_value()) {
539 on_done_(std::move(*status));
540 }
541 }
542
543 // The main body of a step: set the current activity, and any contexts, and
544 // then run the main polling loop. Contained in a function by itself in
545 // order to keep the scoping rules a little easier in Step().
546 absl::optional<ResultType> RunStep() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu()) {
547 ScopedActivity scoped_activity(this);
548 ScopedContext contexts(this);
549 return StepLoop();
550 }
551
552 // Similarly to RunStep, but additionally construct the promise from a
553 // promise factory before entering the main loop. Called once from the
554 // constructor.
555 absl::optional<ResultType> Start(Factory promise_factory)
556 ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu()) {
557 ScopedActivity scoped_activity(this);
558 ScopedContext contexts(this);
559 Construct(&promise_holder_.promise, promise_factory.Make());
560 return StepLoop();
561 }
562
563 // Until there are no wakeups from within and the promise is incomplete:
564 // poll the promise.
565 absl::optional<ResultType> StepLoop() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu()) {
566 GPR_ASSERT(is_current());
567 while (true) {
568 // Run the promise.
569 GPR_ASSERT(!done_);
570 auto r = promise_holder_.promise();
571 if (auto* status = r.value_if_ready()) {
572 // If complete, destroy the promise, flag done, and exit this loop.
573 MarkDone();
574 return IntoStatus(status);
575 }
576 // Continue looping til no wakeups occur.
577 switch (GotActionDuringRun()) {
578 case ActionDuringRun::kNone:
579 return {};
580 case ActionDuringRun::kWakeup:
581 break;
582 case ActionDuringRun::kCancel:
583 MarkDone();
584 return absl::CancelledError();
585 }
586 }
587 }
588
589 using Promise = typename Factory::Promise;
590 // Callback on completion of the promise.
591 GPR_NO_UNIQUE_ADDRESS OnDone on_done_;
592 // Has execution completed?
593 GPR_NO_UNIQUE_ADDRESS bool done_ ABSL_GUARDED_BY(mu()) = false;
594 // Is there a wakeup scheduled?
595 GPR_NO_UNIQUE_ADDRESS std::atomic<bool> wakeup_scheduled_{false};
596 // We wrap the promise in a union to allow control over the construction
597 // simultaneously with annotating mutex requirements and noting that the
598 // promise contained may not use any memory.
599 union PromiseHolder {
600 PromiseHolder() {}
601 ~PromiseHolder() {}
602 GPR_NO_UNIQUE_ADDRESS Promise promise;
603 };
604 GPR_NO_UNIQUE_ADDRESS PromiseHolder promise_holder_ ABSL_GUARDED_BY(mu());
605};
