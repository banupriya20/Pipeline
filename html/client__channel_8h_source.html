73//
74// Client channel filter
75//
76
77// A client channel is a channel that begins disconnected, and can connect
78// to some endpoint on demand. If that endpoint disconnects, it will be
79// connected to again later.
80//
81// Calls on a disconnected client channel are queued until a connection is
82// established.
83
84// Channel arg key for server URI string.
85#define GRPC_ARG_SERVER_URI "grpc.server_uri"
86
87// Channel arg containing a pointer to the ClientChannel object.
88#define GRPC_ARG_CLIENT_CHANNEL "grpc.internal.client_channel"
89
90// Max number of batches that can be pending on a call at any given
91// time. This includes one batch for each of the following ops:
92// recv_initial_metadata
93// send_initial_metadata
94// recv_message
95// send_message
96// recv_trailing_metadata
97// send_trailing_metadata
98#define MAX_PENDING_BATCHES 6

119 // Starts a one-time connectivity state watch. When the channel's state
120 // becomes different from *state, sets *state to the new state and
121 // schedules on_complete. The watcher_timer_init callback is invoked as
122 // soon as the watch is actually started (i.e., after hopping into the
123 // client channel combiner). I/O will be serviced via pollent.
124 //
125 // This is intended to be used when starting a watch from outside of C-core
126 // via grpc_channel_watch_connectivity_state(). It should not be used
127 // by other callers.

130 grpc_closure* on_complete,
131 grpc_closure* watcher_timer_init) {
132 new ExternalConnectivityWatcher(this, pollent, state, on_complete,
133 watcher_timer_init);
