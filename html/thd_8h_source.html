117 Thread& operator=(Thread&& other) noexcept {
118 if (this != &other) {
119 // TODO(vjpai): if we can be sure that all Thread's are actually
120 // constructed, then we should assert GPR_ASSERT(impl_ == nullptr) here.
121 // However, as long as threads come in structures that are
122 // allocated via gpr_malloc, this will not be the case, so we cannot
123 // assert it for the time being.
124 state_ = other.state_;
125 impl_ = other.impl_;
126 options_ = other.options_;
127 other.state_ = MOVED;
128 other.impl_ = nullptr;
129 other.options_ = Options();
130 }
131 return *this;
132 }

139 ~Thread() { GPR_ASSERT(!options_.joinable() || impl_ == nullptr); }
140
141 void Start() {
142 if (impl_ != nullptr) {
143 GPR_ASSERT(state_ == ALIVE);
144 state_ = STARTED;
145 impl_->Start();
146 // If the Thread is not joinable, then the impl_ will cause the deletion
147 // of this Thread object when the thread function completes. Since no
148 // other operation is allowed to a detached thread after Start, there is
149 // no need to change the value of the impl_ or state_ . The next operation
150 // on this object will be the deletion, which will trigger the destructor.
151 } else {
152 GPR_ASSERT(state_ == FAILED);
153 }
154 }
155
156 // It is only legal to call Join if the Thread is created as joinable.
157 void Join() {
158 if (impl_ != nullptr) {
159 impl_->Join();
160 delete impl_;
161 state_ = DONE;
162 impl_ = nullptr;
163 } else {
164 GPR_ASSERT(state_ == FAILED);
165 }
166 }
167
168 private:
169 Thread(const Thread&) = delete;
170 Thread& operator=(const Thread&) = delete;

179 enum ThreadState { FAKE, ALIVE, STARTED, DONE, FAILED, MOVED };
