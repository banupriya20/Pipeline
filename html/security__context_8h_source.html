59#define GRPC_AUTH_CONTEXT_ARG "grpc.auth_context"

61// This type is forward declared as a C struct and we cannot define it as a
62// class. Otherwise, compiler will complain about type mismatch due to
63// -Wmismatched-tags.
64struct grpc_auth_context
65 : public grpc_core::RefCounted<grpc_auth_context,
66 grpc_core::NonPolymorphicRefCount> {
67 public:
68 // Base class for all extensions to inherit from.
69 class Extension {
70 public:
71 virtual ~Extension() = default;
72 };
73 explicit grpc_auth_context(
74 grpc_core::RefCountedPtr<grpc_auth_context> chained)
75 : grpc_core::RefCounted<grpc_auth_context,
76 grpc_core::NonPolymorphicRefCount>(
77 GRPC_TRACE_FLAG_ENABLED(grpc_trace_auth_context_refcount)
78 ? "auth_context_refcount"
79 : nullptr),
80 chained_(std::move(chained)) {
81 if (chained_ != nullptr) {
82 peer_identity_property_name_ = chained_->peer_identity_property_name_;
83 }
84 }
85
86 ~grpc_auth_context() {
87 chained_.reset(DEBUG_LOCATION, "chained");
88 if (properties_.array != nullptr) {
89 for (size_t i = 0; i < properties_.count; i++) {
90 grpc_auth_property_reset(&properties_.array[i]);
91 }
92 gpr_free(properties_.array);
93 }
94 }
95
96 static absl::string_view ChannelArgName() { return GRPC_AUTH_CONTEXT_ARG; }
97 static int ChannelArgsCompare(const grpc_auth_context* a,
98 const grpc_auth_context* b) {
99 return QsortCompare(a, b);
100 }
101
102 const grpc_auth_context* chained() const { return chained_.get(); }
103 const grpc_auth_property_array& properties() const { return properties_; }
104
105 bool is_authenticated() const {
106 return peer_identity_property_name_ != nullptr;
107 }
108 const char* peer_identity_property_name() const {
109 return peer_identity_property_name_;
110 }
111 void set_peer_identity_property_name(const char* name) {
112 peer_identity_property_name_ = name;
113 }
114 void set_extension(std::unique_ptr<Extension> extension) {
115 extension_ = std::move(extension);
116 }
117
118 void ensure_capacity();
119 void add_property(const char* name, const char* value, size_t value_length);
120 void add_cstring_property(const char* name, const char* value);
121
122 private:
123 grpc_core::RefCountedPtr<grpc_auth_context> chained_;
124 grpc_auth_property_array properties_;
125 const char* peer_identity_property_name_ = nullptr;
126 std::unique_ptr<Extension> extension_;
127};
