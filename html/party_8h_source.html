42// Two implementations of party synchronization are provided: one using a single
43// atomic, the other using a mutex and a set of state variables.
44// Originally the atomic implementation was implemented, but we found some race
45// conditions on Arm that were not reported by our default TSAN implementation.
46// The mutex implementation was added to see if it would fix the problem, and
47// it did. Later we found the race condition, so there's no known reason to use
48// the mutex version - however we keep it around as a just in case measure.
49// There's a thought of fuzzing the two implementations against each other as
50// a correctness check of both, but that's not implemented yet.
51

53// #define GRPC_PARTY_SYNC_USING_MUTEX

56 defined(GRPC_PARTY_SYNC_USING_MUTEX) != \
57 1
58#error Must define a party sync mechanism

71class PartySyncUsingAtomics {
72 public:
73 explicit PartySyncUsingAtomics(size_t initial_refs)
74 : state_(kOneRef * initial_refs) {}
75
76 void IncrementRefCount() {
77 state_.fetch_add(kOneRef, std::memory_order_relaxed);
78 }
79 GRPC_MUST_USE_RESULT bool RefIfNonZero();
80 // Returns true if the ref count is now zero and the caller should call
81 // PartyOver
82 GRPC_MUST_USE_RESULT bool Unref() {
83 uint64_t prev_state = state_.fetch_sub(kOneRef, std::memory_order_acq_rel);
84 if ((prev_state & kRefMask) == kOneRef) {
85 return UnreffedLast();
86 }
87 return false;
88 }
89 void ForceImmediateRepoll(WakeupMask mask) {
90 // Or in the bit for the currently polling participant.
91 // Will be grabbed next round to force a repoll of this promise.
92 state_.fetch_or(mask, std::memory_order_relaxed);
93 }
94
95 // Run the update loop: poll_one_participant is called with an integral index
96 // for the participant that should be polled. It should return true if the
97 // participant completed and should be removed from the allocated set.
98 template <typename F>
99 GRPC_MUST_USE_RESULT bool RunParty(F poll_one_participant) {
100 uint64_t prev_state;
101 do {
102 // Grab the current state, and clear the wakeup bits & add flag.
103 prev_state = state_.fetch_and(kRefMask | kLocked | kAllocatedMask,
104 std::memory_order_acquire);
105 GPR_ASSERT(prev_state & kLocked);
106 if (prev_state & kDestroying) return true;
107 // From the previous state, extract which participants we're to wakeup.
108 uint64_t wakeups = prev_state & kWakeupMask;
109 // Now update prev_state to be what we want the CAS to see below.
110 prev_state &= kRefMask | kLocked | kAllocatedMask;
111 // For each wakeup bit...
112 for (size_t i = 0; wakeups != 0; i++, wakeups >>= 1) {
113 // If the bit is not set, skip.
114 if ((wakeups & 1) == 0) continue;
115 if (poll_one_participant(i)) {
116 const uint64_t allocated_bit = (1u << i << kAllocatedShift);
117 prev_state &= ~allocated_bit;
118 state_.fetch_and(~allocated_bit, std::memory_order_release);
119 }
120 }
121 // Try to CAS the state we expected to have (with no wakeups or adds)
122 // back to unlocked (by masking in only the ref mask - sans locked bit).
123 // If this succeeds then no wakeups were added, no adds were added, and we
124 // have successfully unlocked.
125 // Otherwise, we need to loop again.
126 // Note that if an owning waker is created or the weak cas spuriously
127 // fails we will also loop again, but in that case see no wakeups or adds
128 // and so will get back here fairly quickly.
129 // TODO(ctiller): consider mitigations for the accidental wakeup on owning
130 // waker creation case -- I currently expect this will be more expensive
131 // than this quick loop.
132 } while (!state_.compare_exchange_weak(
133 prev_state, (prev_state & (kRefMask | kAllocatedMask)),
134 std::memory_order_acq_rel, std::memory_order_acquire));
135 return false;
136 }
137
138 // Add new participants to the party. Returns true if the caller should run
139 // the party. store is called with an array of indices of the new
140 // participants. Adds a ref that should be dropped by the caller after
141 // RunParty has been called (if that was required).
142 template <typename F>
143 GRPC_MUST_USE_RESULT bool AddParticipantsAndRef(size_t count, F store) {
144 uint64_t state = state_.load(std::memory_order_acquire);
145 uint64_t allocated;
146
147 size_t slots[party_detail::kMaxParticipants];
148
149 // Find slots for each new participant, ordering them from lowest available
150 // slot upwards to ensure the same poll ordering as presentation ordering to
151 // this function.
152 WakeupMask wakeup_mask;
153 do {
154 wakeup_mask = 0;
155 allocated = (state & kAllocatedMask) >> kAllocatedShift;
156 size_t n = 0;
157 for (size_t bit = 0; n < count && bit < party_detail::kMaxParticipants;
158 bit++) {
159 if (allocated & (1 << bit)) continue;
160 wakeup_mask |= (1 << bit);
161 slots[n++] = bit;
162 allocated |= 1 << bit;
163 }
164 GPR_ASSERT(n == count);
165 // Try to allocate this slot and take a ref (atomically).
166 // Ref needs to be taken because once we store the participant it could be
167 // spuriously woken up and unref the party.
168 } while (!state_.compare_exchange_weak(
169 state, (state | (allocated << kAllocatedShift)) + kOneRef,
170 std::memory_order_acq_rel, std::memory_order_acquire));
171
172 store(slots);
173
174 // Now we need to wake up the party.
175 state = state_.fetch_or(wakeup_mask | kLocked, std::memory_order_release);
176
177 // If the party was already locked, we're done.
178 return ((state & kLocked) == 0);
179 }
180
181 // Schedule a wakeup for the given participant.
182 // Returns true if the caller should run the party.
183 GRPC_MUST_USE_RESULT bool ScheduleWakeup(WakeupMask mask);
184
185 private:
186 bool UnreffedLast();
187
188 // State bits:
189 // The atomic state_ field is composed of the following:
190 // - 24 bits for ref counts
191 // 1 is owned by the party prior to Orphan()
192 // All others are owned by owning wakers
193 // - 1 bit to indicate whether the party is locked
194 // The first thread to set this owns the party until it is unlocked
195 // That thread will run the main loop until no further work needs to
196 // be done.
197 // - 1 bit to indicate whether there are participants waiting to be
198 // added
199 // - 16 bits, one per participant, indicating which participants have
200 // been
201 // woken up and should be polled next time the main loop runs.
202
203 // clang-format off
204 // Bits used to store 16 bits of wakeups
205 static constexpr uint64_t kWakeupMask = 0x0000'0000'0000'ffff;
206 // Bits used to store 16 bits of allocated participant slots.
207 static constexpr uint64_t kAllocatedMask = 0x0000'0000'ffff'0000;
208 // Bit indicating destruction has begun (refs went to zero)
209 static constexpr uint64_t kDestroying = 0x0000'0001'0000'0000;
210 // Bit indicating locked or not
211 static constexpr uint64_t kLocked = 0x0000'0008'0000'0000;
212 // Bits used to store 24 bits of ref counts
213 static constexpr uint64_t kRefMask = 0xffff'ff00'0000'0000;
214 // clang-format on
215
216 // Shift to get from a participant mask to an allocated mask.
217 static constexpr size_t kAllocatedShift = 16;
218 // How far to shift to get the refcount
219 static constexpr size_t kRefShift = 40;
220 // One ref count
221 static constexpr uint64_t kOneRef = 1ull << kRefShift;
222
223 std::atomic<uint64_t> state_;
224};

226class PartySyncUsingMutex {
227 public:
228 explicit PartySyncUsingMutex(size_t initial_refs) : refs_(initial_refs) {}
229
230 void IncrementRefCount() { refs_.Ref(); }
231 GRPC_MUST_USE_RESULT bool RefIfNonZero() { return refs_.RefIfNonZero(); }
232 GRPC_MUST_USE_RESULT bool Unref() { return refs_.Unref(); }
233 void ForceImmediateRepoll(WakeupMask mask) {
234 MutexLock lock(&mu_);
235 wakeups_ |= mask;
236 }
237 template <typename F>
238 GRPC_MUST_USE_RESULT bool RunParty(F poll_one_participant) {
239 WakeupMask freed = 0;
240 while (true) {
241 ReleasableMutexLock lock(&mu_);
242 GPR_ASSERT(locked_);
243 allocated_ &= ~std::exchange(freed, 0);
244 auto wakeup = std::exchange(wakeups_, 0);
245 if (wakeup == 0) {
246 locked_ = false;
247 return false;
248 }
249 lock.Release();
250 for (size_t i = 0; wakeup != 0; i++, wakeup >>= 1) {
251 if ((wakeup & 1) == 0) continue;
252 if (poll_one_participant(i)) freed |= 1 << i;
253 }
254 }
255 }
256
257 template <typename F>
258 GRPC_MUST_USE_RESULT bool AddParticipantsAndRef(size_t count, F store) {
259 IncrementRefCount();
260 MutexLock lock(&mu_);
261 size_t slots[party_detail::kMaxParticipants];
262 WakeupMask wakeup_mask = 0;
263 size_t n = 0;
264 for (size_t bit = 0; n < count && bit < party_detail::kMaxParticipants;
265 bit++) {
266 if (allocated_ & (1 << bit)) continue;
267 slots[n++] = bit;
268 wakeup_mask |= 1 << bit;
269 allocated_ |= 1 << bit;
270 }
271 GPR_ASSERT(n == count);
272 store(slots);
273 wakeups_ |= wakeup_mask;
274 return !std::exchange(locked_, true);
275 }
276
277 GRPC_MUST_USE_RESULT bool ScheduleWakeup(WakeupMask mask);
278
279 private:
280 RefCount refs_;
281 Mutex mu_;
282 WakeupMask allocated_ ABSL_GUARDED_BY(mu_) = 0;
283 WakeupMask wakeups_ ABSL_GUARDED_BY(mu_) = 0;
284 bool locked_ ABSL_GUARDED_BY(mu_) = false;
285};

376 // Main run loop. Must be locked.
377 // Polls participants and drains the add queue until there is no work left to
378 // be done.
379 // Derived types will likely want to override this to set up their
380 // contexts before polling.
381 // Should not be called by derived types except as a tail call to the base

383 // Returns true if the party is over.

394 // Concrete implementation of a participant for some promise & oncomplete
395 // type.
396 template <typename SuppliedFactory, typename OnComplete>
397 class ParticipantImpl final : public Participant {
398 using Factory = promise_detail::OncePromiseFactory<void, SuppliedFactory>;
399 using Promise = typename Factory::Promise;
400
401 public:
402 ParticipantImpl(absl::string_view name, SuppliedFactory promise_factory,
403 OnComplete on_complete)
404 : Participant(name), on_complete_(std::move(on_complete)) {
405 Construct(&factory_, std::move(promise_factory));
406 }
407 ~ParticipantImpl() {
408 if (!started_) {
409 Destruct(&factory_);
410 } else {
411 Destruct(&promise_);
412 }
413 }
414
415 bool Poll() override {
416 if (!started_) {
417 auto p = factory_.Make();
418 Destruct(&factory_);
419 Construct(&promise_, std::move(p));
420 started_ = true;
421 }
422 auto p = promise_();
423 if (auto* r = p.value_if_ready()) {
424 on_complete_(std::move(*r));
425 GetContext<Arena>()->DeletePooled(this);
426 return true;
427 }
428 return false;
429 }
430
431 void Destroy() override { GetContext<Arena>()->DeletePooled(this); }
