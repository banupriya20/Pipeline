51 ParsedMetadata<grpc_metadata_batch> md;
52 absl::Status parse_status;

55 // Lookup, but don't ref.
56 const Memento* Lookup(uint32_t index) const {
57 // Static table comes first, just return an entry from it.
58 // NB: This imposes the constraint that the first
59 // GRPC_CHTTP2_LAST_STATIC_ENTRY entries in the core static metadata table
60 // must follow the hpack standard. If that changes, we *must* not rely on
61 // reading the core static metadata table here; at that point we'd need our
62 // own singleton static metadata in the correct order.
63 if (index <= hpack_constants::kLastStaticEntry) {
64 return &static_mementos_->memento[index - 1];
65 } else {
66 return LookupDynamic(index);
67 }
68 }
69
70 // add a table entry to the index
71 grpc_error_handle Add(Memento md) GRPC_MUST_USE_RESULT;
72
73 // Current entry count in the table.
74 uint32_t num_entries() const { return entries_.num_entries(); }
75
76 // Current size of the table.
77 uint32_t test_only_table_size() const { return mem_used_; }
78
79 private:
80 struct StaticMementos {
81 StaticMementos();
82 Memento memento[hpack_constants::kLastStaticEntry];
83 };

85 class MementoRingBuffer {
86 public:
87 // Rebuild this buffer with a new max_entries_ size.
88 void Rebuild(uint32_t max_entries);
89
90 // Put a new memento.
91 // REQUIRES: num_entries < max_entries
92 void Put(Memento m);
93
94 // Pop the oldest memento.
95 // REQUIRES: num_entries > 0
96 Memento PopOne();
97
98 // Lookup the entry at index, or return nullptr if none exists.
99 const Memento* Lookup(uint32_t index) const;
100
101 uint32_t max_entries() const { return max_entries_; }
102 uint32_t num_entries() const { return num_entries_; }
103
104 private:
105 // The index of the first entry in the buffer. May be greater than
106 // max_entries_, in which case a wraparound has occurred.
107 uint32_t first_entry_ = 0;
108 // How many entries are in the table.
109 uint32_t num_entries_ = 0;
110 // Maximum number of entries we could possibly fit in the table, given
111 // defined overheads.
112 uint32_t max_entries_ = hpack_constants::kInitialTableEntries;
113
114 std::vector<Memento> entries_;
115 };

117 const Memento* LookupDynamic(uint32_t index) const {
118 // Not static - find the value in the list of valid entries
119 const uint32_t tbl_index = index - (hpack_constants::kLastStaticEntry + 1);
120 return entries_.Lookup(tbl_index);
121 }

126 static const NoDestruct<StaticMementos> static_mementos;
127 return static_mementos.get();

130 // The amount of memory used by the table, according to the hpack algorithm
131 uint32_t mem_used_ = 0;
132 // The max memory allowed to be used by the table, according to the hpack
133 // algorithm.
134 uint32_t max_bytes_ = hpack_constants::kInitialTableSize;
135 // The currently agreed size of the table, according to the hpack algorithm.
136 uint32_t current_table_bytes_ = hpack_constants::kInitialTableSize;
137 // HPack table entries
138 MementoRingBuffer entries_;
139 // Static mementos
140 const StaticMementos* const static_mementos_ = GetStaticMementos();
141};
