59// Define a traits object for vtable lookup - allows us to integrate with
60// existing code easily (just define the trait!) and allows some magic in
61// ChannelArgs to automatically derive a vtable from a T*.
62// To participate as a pointer, instances should expose the function:
63// // Gets the vtable for this type
64// static const grpc_arg_pointer_vtable* VTable();
65// // Performs any mutations required for channel args to own a pointer
66// // Only needed if ChannelArgs::Set is to be called with a raw pointer.
67// static void* TakeUnownedPointer(T* p);
68template <typename T, typename Ignored = void /* for SFINAE */>
69struct ChannelArgTypeTraits;

71namespace channel_args_detail {
72inline int PointerCompare(void* a_ptr, const grpc_arg_pointer_vtable* a_vtable,
73 void* b_ptr,
74 const grpc_arg_pointer_vtable* b_vtable) {
75 int c = QsortCompare(a_ptr, b_ptr);
76 if (c == 0) return 0;
77 c = QsortCompare(a_vtable, b_vtable);
78 if (c != 0) return c;
79 return a_vtable->cmp(a_ptr, b_ptr);
80}

88// Specialization for ref-counted pointers.
89// Types should expose:
90// static int ChannelArgsCompare(const T* a, const T* b);
91template <typename T>
92struct ChannelArgTypeTraits<
93 T, absl::enable_if_t<
94 std::is_base_of<RefCounted<channel_args_detail::RefType<T>>,
95 channel_args_detail::RefType<T>>::value ||
96 std::is_base_of<RefCounted<channel_args_detail::RefType<T>,
97 NonPolymorphicRefCount>,
98 channel_args_detail::RefType<T>>::value ||
99 std::is_base_of<DualRefCounted<channel_args_detail::RefType<T>>,
100 channel_args_detail::RefType<T>>::value,
101 void>> {
102 static const grpc_arg_pointer_vtable* VTable() {
103 static const grpc_arg_pointer_vtable tbl = {
104 // copy
105 [](void* p) -> void* {
106 return p == nullptr ? nullptr
107 : static_cast<T*>(p)
108 ->Ref(DEBUG_LOCATION, "ChannelArgs copy")
109 .release();
110 },
111 // destroy
112 [](void* p) {
113 if (p != nullptr) {
114 static_cast<T*>(p)->Unref(DEBUG_LOCATION, "ChannelArgs destroy");
115 }
116 },
117 // compare
118 [](void* p1, void* p2) {
119 return T::ChannelArgsCompare(static_cast<const T*>(p1),
120 static_cast<const T*>(p2));
121 },
122 };
123 return &tbl;
124 };
125};

127// Specialization for shared_ptr
128// Incurs an allocation because shared_ptr.release is not a thing.
129template <typename T>
130struct is_shared_ptr : std::false_type {};
131template <typename T>
132struct is_shared_ptr<std::shared_ptr<T>> : std::true_type {};
133template <typename T>
134struct ChannelArgTypeTraits<T,
135 absl::enable_if_t<is_shared_ptr<T>::value, void>> {
136 static void* TakeUnownedPointer(T* p) { return p; }
137 static const grpc_arg_pointer_vtable* VTable() {
138 static const grpc_arg_pointer_vtable tbl = {
139 // copy
140 [](void* p) -> void* { return new T(*static_cast<T*>(p)); },
141 // destroy
142 [](void* p) { delete static_cast<T*>(p); },
143 // compare
144 [](void* p1, void* p2) {
145 return QsortCompare(static_cast<const T*>(p1)->get(),
146 static_cast<const T*>(p2)->get());
147 },
148 };
149 return &tbl;
150 };
151};

153// If a type declares some member 'struct RawPointerChannelArgTag {}' then
154// we automatically generate a vtable for it that does not do any ownership
155// management and compares the type by pointer identity.
156// This is intended to be relatively ugly because *most types should worry about
157// ownership*.
158template <typename T>
159struct ChannelArgTypeTraits<T,
160 absl::void_t<typename T::RawPointerChannelArgTag>> {
161 static void* TakeUnownedPointer(T* p) { return p; }
162 static const grpc_arg_pointer_vtable* VTable() {
163 static const grpc_arg_pointer_vtable tbl = {
164 // copy
165 [](void* p) -> void* { return p; },
166 // destroy
167 [](void*) {},
168 // compare
169 [](void* p1, void* p2) { return QsortCompare(p1, p2); },
170 };
171 return &tbl;
172 };
173};

183template <typename T, typename Ignored = void /* for SFINAE */>
184struct GetObjectImpl;
185// std::shared_ptr implementation
186template <typename T>
187struct GetObjectImpl<T, absl::enable_if_t<WrapInSharedPtr<T>::value, void>> {
188 using Result = T*;
189 using ReffedResult = std::shared_ptr<T>;
190 using StoredType = std::shared_ptr<T>*;
191 static Result Get(StoredType p) {
192 if (p == nullptr) return nullptr;
193 return p->get();
194 };
195 static ReffedResult GetReffed(StoredType p) {
196 if (p == nullptr) return nullptr;
197 return ReffedResult(*p);
198 };
199 static ReffedResult GetReffed(StoredType p,
200 const DebugLocation& /* location */,
201 const char* /* reason */) {
202 return GetReffed(*p);
203 };
204};
205// RefCountedPtr
206template <typename T>
207struct GetObjectImpl<T, absl::enable_if_t<!WrapInSharedPtr<T>::value, void>> {
208 using Result = T*;
209 using ReffedResult = RefCountedPtr<T>;
210 using StoredType = Result;
211 static Result Get(StoredType p) { return p; };
212 static ReffedResult GetReffed(StoredType p) {
213 if (p == nullptr) return nullptr;
214 return p->Ref();
215 };
216 static ReffedResult GetReffed(StoredType p, const DebugLocation& location,
217 const char* reason) {
218 if (p == nullptr) return nullptr;
219 return p->Ref(location, reason);
220 };
221};
