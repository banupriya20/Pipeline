146// Registers various things for the OpenCensus plugin.
147class OpenCensusRegistry {
148 public:
149 struct Label {
150 std::string key;
151 opencensus::tags::TagKey tag_key;
152 std::string value;
153 };
154
155 struct Attribute {
156 std::string key;
157 std::string value;
158 };
159
160 static OpenCensusRegistry& Get();
161
162 void RegisterConstantLabels(
163 const std::map<std::string, std::string>& labels) {
164 constant_labels_.reserve(labels.size());
165 for (const auto& label : labels) {
166 auto tag_key = opencensus::tags::TagKey::Register(label.first);
167 constant_labels_.emplace_back(Label{label.first, tag_key, label.second});
168 }
169 }
170
171 void RegisterConstantAttributes(std::vector<Attribute> attributes) {
172 constant_attributes_ = std::move(attributes);
173 }
174
175 ::opencensus::tags::TagMap PopulateTagMapWithConstantLabels(
176 const ::opencensus::tags::TagMap& tag_map);
177
178 void PopulateCensusContextWithConstantAttributes(
179 grpc::experimental::CensusContext* context);
180
181 const std::vector<Label>& ConstantLabels() { return constant_labels_; }
182
183 const std::vector<Attribute>& ConstantAttributes() {
184 return constant_attributes_;
185 }
186
187 private:
188 OpenCensusRegistry() = default;
189
190 std::vector<Label> constant_labels_;
191 std::vector<Attribute> constant_attributes_;
192};
