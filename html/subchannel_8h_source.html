159// A subchannel that knows how to connect to exactly one target address. It
160// provides a target for load balancing.
161//
162// Note that this is the "real" subchannel implementation, whose API is
163// different from the SubchannelInterface that is exposed to LB policy
164// implementations. The client channel provides an adaptor class
165// (SubchannelWrapper) that "converts" between the two.
166class Subchannel : public DualRefCounted<Subchannel> {
167 public:
168 // TODO(roth): Once we remove pollset_set, consider whether this can
169 // just use the normal AsyncConnectivityStateWatcherInterface API.
170 class ConnectivityStateWatcherInterface
171 : public RefCounted<ConnectivityStateWatcherInterface> {
172 public:
173 // Invoked whenever the subchannel's connectivity state changes.
174 // There will be only one invocation of this method on a given watcher
175 // instance at any given time.
176 virtual void OnConnectivityStateChange(grpc_connectivity_state state,
177 const absl::Status& status) = 0;
178
179 virtual grpc_pollset_set* interested_parties() = 0;
180 };

182 // A base class for producers of subchannel-specific data.
183 // Implementations will typically add their own methods as needed.
184 class DataProducerInterface : public DualRefCounted<DataProducerInterface> {
185 public:
186 // A unique identifier for this implementation.
187 // Only one producer may be registered under a given type name on a
188 // given subchannel at any given time.
189 // Note that we use the pointer address instead of the string
190 // contents for uniqueness; all instances for a given implementation
191 // are expected to return the same string *instance*, not just the
192 // same string contents.
193 virtual UniqueTypeName type() const = 0;
194 };

216 // The first callback to the watcher will be delivered ~immediately.

245 // Access to data producer map.
246 // We do not hold refs to the data producer; the implementation is
247 // expected to register itself upon construction and remove itself
248 // upon destruction.
249 //
250 // Looks up the current data producer for type and invokes get_or_add()
251 // with a pointer to that producer in the map. The get_or_add() function
252 // can modify the pointed-to value to update the map. This provides a
253 // way to either re-use an existing producer or register a new one in
254 // a non-racy way.
255 void GetOrAddDataProducer(
256 UniqueTypeName type,
257 std::function<void(DataProducerInterface**)> get_or_add)
258 ABSL_LOCKS_EXCLUDED(mu_);
259 // Removes the data producer from the map, if the current producer for
260 // this type is the specified producer.
261 void RemoveDataProducer(DataProducerInterface* data_producer)
262 ABSL_LOCKS_EXCLUDED(mu_);
