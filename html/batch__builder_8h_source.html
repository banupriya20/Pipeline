55// Build up a transport stream op batch for a stream for a promise based
56// connected channel.
57// Offered as a context from Call, so that it can collect ALL the updates during
58// a single party round, and then push them down to the transport as a single
59// transaction.
60class BatchBuilder {
61 public:
62 explicit BatchBuilder(grpc_transport_stream_op_batch_payload* payload);
63 ~BatchBuilder() {
64 if (batch_ != nullptr) FlushBatch();
65 }
66
67 struct Target {
68 grpc_transport* transport;
69 grpc_stream* stream;
70 grpc_stream_refcount* stream_refcount;
71 };

198 // One outstanding batch.
199 struct Batch final {
200 Batch(grpc_transport_stream_op_batch_payload* payload,
201 grpc_stream_refcount* stream_refcount);
202 ~Batch();
203 Batch(const Batch&) = delete;
204 Batch& operator=(const Batch&) = delete;
205 void IncrementRefCount() { ++refs; }
206 void Unref() {
207 if (--refs == 0) party->arena()->DeletePooled(this);
208 }
209 RefCountedPtr<Batch> Ref() {
210 IncrementRefCount();
211 return RefCountedPtr<Batch>(this);
212 }
213 // Get an initialized pending completion.
214 // There are four pending completions potentially contained within a batch.
215 // They can be rather large so we don't create all of them always. Instead,
216 // we dynamically create them on the arena as needed.
217 // This method either returns the existing completion in a batch if that
218 // completion has already been initialized, or it creates a new completion
219 // and returns that.
220 template <typename T>
221 T* GetInitializedCompletion(T*(Batch::*field)) {
222 if (this->*field != nullptr) return this->*field;
223 this->*field = party->arena()->NewPooled<T>(Ref());
224 if (grpc_call_trace.enabled()) {
225 gpr_log(GPR_DEBUG, "%s[connected] Add batch closure for %s @ %s",
226 Activity::current()->DebugTag().c_str(),
227 std::string((this->*field)->name()).c_str(),
228 (this->*field)->on_done_closure.DebugString().c_str());
229 }
230 return this->*field;
231 }
232 // grpc_transport_perform_stream_op on target.stream
233 void PerformWith(Target target);
234 // Take a promise, and return a promise that holds a ref on this batch until
235 // the promise completes or is cancelled.
236 template <typename P>
237 auto RefUntil(P promise) {
238 return [self = Ref(), promise = std::move(promise)]() mutable {
239 return promise();
240 };
241 }
242
243 grpc_transport_stream_op_batch batch;
244 PendingReceiveMessage* pending_receive_message = nullptr;
245 PendingReceiveInitialMetadata* pending_receive_initial_metadata = nullptr;
246 PendingReceiveTrailingMetadata* pending_receive_trailing_metadata = nullptr;
247 PendingSends* pending_sends = nullptr;
248 const RefCountedPtr<Party> party;
249 grpc_stream_refcount* const stream_refcount;
250 uint8_t refs = 0;
251 };

253 // Get a batch for the given target.
254 // Currently: if the current batch is for this target, return it - otherwise
255 // flush the batch and start a new one (and return that).
256 // This function may change in the future to allow multiple batches to be
257 // building at once (if that turns out to be useful for hedging).
258 Batch* GetBatch(Target target);
259 // Flush the current batch down to the transport.
260 void FlushBatch();
261 // Create a cancel batch with its own payload.
262 Batch* MakeCancel(grpc_stream_refcount* stream_refcount, absl::Status status);

284inline auto BatchBuilder::SendMessage(Target target, MessageHandle message) {
285 if (grpc_call_trace.enabled()) {
286 gpr_log(GPR_DEBUG, "%s[connected] Queue send message: %s",
287 Activity::current()->DebugTag().c_str(),
288 message->DebugString().c_str());
289 }
290 auto* batch = GetBatch(target);
291 auto* pc = batch->GetInitializedCompletion(&Batch::pending_sends);
292 batch->batch.on_complete = &pc->on_done_closure;
293 batch->batch.send_message = true;
294 payload_->send_message.send_message = message->payload();
295 payload_->send_message.flags = message->flags();
296 pc->send_message = std::move(message);
297 return batch->RefUntil(pc->done_latch.WaitAndCopy());
298}

342inline auto BatchBuilder::SendServerTrailingMetadata(
343 Target target, ServerMetadataHandle metadata,
344 bool convert_to_cancellation) {
345 if (grpc_call_trace.enabled()) {
346 gpr_log(GPR_DEBUG, "%s[connected] %s: %s",
347 Activity::current()->DebugTag().c_str(),
348 convert_to_cancellation ? "Send trailing metadata as cancellation"
349 : "Queue send trailing metadata",
350 metadata->DebugString().c_str());
351 }
352 Batch* batch;
353 PendingSends* pc;
354 if (convert_to_cancellation) {
355 const auto status_code =
356 metadata->get(GrpcStatusMetadata()).value_or(GRPC_STATUS_UNKNOWN);
357 auto status = grpc_error_set_int(
358 absl::Status(static_cast<absl::StatusCode>(status_code),
359 metadata->GetOrCreatePointer(GrpcMessageMetadata())
360 ->as_string_view()),
361 StatusIntProperty::kRpcStatus, status_code);
362 batch = MakeCancel(target.stream_refcount, std::move(status));
363 pc = batch->GetInitializedCompletion(&Batch::pending_sends);
364 } else {
365 batch = GetBatch(target);
366 pc = batch->GetInitializedCompletion(&Batch::pending_sends);
367 batch->batch.send_trailing_metadata = true;
368 payload_->send_trailing_metadata.send_trailing_metadata = metadata.get();
369 payload_->send_trailing_metadata.sent = &pc->trailing_metadata_sent;
370 }
371 batch->batch.on_complete = &pc->on_done_closure;
372 pc->send_trailing_metadata = std::move(metadata);
373 auto promise = batch->RefUntil(
374 Map(pc->done_latch.WaitAndCopy(), [pc](absl::Status status) {
375 return CompleteSendServerTrailingMetadata(
376 std::move(pc->send_trailing_metadata), std::move(status),
377 pc->trailing_metadata_sent);
378 }));
379 if (convert_to_cancellation) {
380 batch->PerformWith(target);
381 }
382 return promise;
383}

385inline auto BatchBuilder::ReceiveMessage(Target target) {
386 if (grpc_call_trace.enabled()) {
387 gpr_log(GPR_DEBUG, "%s[connected] Queue receive message",
388 Activity::current()->DebugTag().c_str());
389 }
390 auto* batch = GetBatch(target);
391 auto* pc = batch->GetInitializedCompletion(&Batch::pending_receive_message);
392 batch->batch.recv_message = true;
393 payload_->recv_message.recv_message_ready = &pc->on_done_closure;
394 payload_->recv_message.recv_message = &pc->payload;
395 payload_->recv_message.flags = &pc->flags;
396 payload_->recv_message.call_failed_before_recv_message =
397 &pc->call_failed_before_recv_message;
398 return batch->RefUntil(
399 Map(pc->done_latch.Wait(),
400 [pc](absl::Status status)
401 -> absl::StatusOr<absl::optional<MessageHandle>> {
402 if (!status.ok()) return status;
403 if (!pc->payload.has_value()) {
404 if (pc->call_failed_before_recv_message) {
405 return absl::CancelledError();
406 }
407 return absl::nullopt;
408 }
409 return pc->IntoMessageHandle();
410 }));
411}
