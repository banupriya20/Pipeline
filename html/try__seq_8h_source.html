75// If there exists a function 'IsStatusOk(const T&) -> bool' then we assume that
76// T is a status type for the purposes of promise sequences, and a non-OK T
77// should terminate the sequence and return.
78template <typename T>
79struct TrySeqTraitsWithSfinae<
80 T, absl::enable_if_t<
81 std::is_same<decltype(IsStatusOk(std::declval<T>())), bool>::value,
82 void>> {
83 using UnwrappedType = void;
84 using WrappedType = T;
85 template <typename Next>
86 static auto CallFactory(Next* next, T&&) {
87 return next->Make();
88 }
89 template <typename Result, typename RunNext>
90 static Poll<Result> CheckResultAndRunNext(T prior, RunNext run_next) {
91 if (!IsStatusOk(prior)) return Result(std::move(prior));
92 return run_next(std::move(prior));
93 }
94};

139// Try a sequence of operations.
140// * Run the first functor as a promise.
141// * Feed its success result into the second functor to create a promise,
142// then run that.
143// * ...
144// * Feed the second-final success result into the final functor to create a
145// promise, then run that, with the overall success result being that
146// promises success result.
147// If any step fails, fail everything.
148// Functors can return StatusOr<> to signal that a value is fed forward, or
149// Status to indicate only success/failure. In the case of returning Status,
150// the construction functors take no arguments.
151template <typename... Functors>
152promise_detail::TrySeq<Functors...> TrySeq(Functors... functors) {
153 return promise_detail::TrySeq<Functors...>(std::move(functors)...);
154}
155
156// Try a sequence of operations of unknown length.
157// Asynchronously:
158// for (element in (begin, end)) {
159// auto r = wait_for factory(element, argument);
160// if (!r.ok()) return r;
161// argument = *r;
162// }
163// return argument;
164template <typename Iter, typename Factory, typename Argument>
165typename promise_detail::TrySeqIterResultTraits<Iter, Factory, Argument>::Result
166TrySeqIter(Iter begin, Iter end, Argument argument, Factory factory) {
167 using Result =
168 typename promise_detail::TrySeqIterResultTraits<Iter, Factory,
169 Argument>::Result;
170 return Result(begin, end, std::move(factory), std::move(argument));
171}
