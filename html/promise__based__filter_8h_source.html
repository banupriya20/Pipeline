223 void Resume(grpc_transport_stream_op_batch* batch) {
224 GPR_ASSERT(!call_->is_last());

229 }

239 call_closures_.Add(batch->on_complete, absl::OkStatus(),
240 "Flusher::Complete");
241 }

244 const char* reason) {
245 call_closures_.Add(closure, error, reason);
246 }

347 // State machine for sending messages: handles intercepting send_message ops
348 // and forwarding them through pipes to the promise, then getting the result
349 // down the stack.
350 // Split into its own class so that we don't spend the memory instantiating
351 // these members for filters that don't need to intercept sent messages.
352 class SendMessage {
353 public:
354 SendMessage(BaseCallData* base, Interceptor* interceptor)
355 : base_(base), interceptor_(interceptor) {}
356 ~SendMessage() { interceptor_->~Interceptor(); }
357
358 Interceptor* interceptor() { return interceptor_; }
359
360 // Start a send_message op.
361 void StartOp(CapturedBatch batch);
362 // Publish the outbound pipe to the filter.
363 // This happens when the promise requests to call the next filter: until
364 // this occurs messages can't be sent as we don't know the pipe that the
365 // promise expects to send on.
366 template <typename T>
367 void GotPipe(T* pipe);
368 // Called from client/server polling to do the send message part of the
369 // work.
370 void WakeInsideCombiner(Flusher* flusher, bool allow_push_to_pipe);
371 // Call is completed, we have trailing metadata. Close things out.
372 void Done(const ServerMetadata& metadata, Flusher* flusher);
373 // Return true if we have a batch captured (for debug logs)
374 bool HaveCapturedBatch() const { return batch_.is_captured(); }
375 // Return true if we're not actively sending a message.
376 bool IsIdle() const;
377 // Return true if we've released the message for forwarding down the stack.
378 bool IsForwarded() const { return state_ == State::kForwardedBatch; }
379
380 private:
381 enum class State : uint8_t {
382 // Starting state: no batch started, no outgoing pipe configured.
383 kInitial,
384 // We have an outgoing pipe, but no batch started.
385 // (this is the steady state).
386 kIdle,
387 // We have a batch started, but no outgoing pipe configured.
388 // Stall until we have one.
389 kGotBatchNoPipe,
390 // We have a batch, and an outgoing pipe. On the next poll we'll push the
391 // message into the pipe to the promise.
392 kGotBatch,
393 // We've pushed a message into the promise, and we're now waiting for it
394 // to pop out the other end so we can forward it down the stack.
395 kPushedToPipe,
396 // We've forwarded a message down the stack, and now we're waiting for
397 // completion.
398 kForwardedBatch,
399 // We've got the completion callback, we'll close things out during poll
400 // and then forward completion callbacks up and transition back to idle.
401 kBatchCompleted,
402 // We're almost done, but need to poll first.
403 kCancelledButNotYetPolled,
404 // We're done.
405 kCancelled,
406 // We're done, but we haven't gotten a status yet
407 kCancelledButNoStatus,
408 };
409 static const char* StateString(State);
410
411 void OnComplete(absl::Status status);
412
413 BaseCallData* const base_;
414 State state_ = State::kInitial;
415 Interceptor* const interceptor_;
416 absl::optional<PipeSender<MessageHandle>::PushType> push_;
417 absl::optional<PipeReceiverNextType<MessageHandle>> next_;
418 CapturedBatch batch_;
419 grpc_closure* intercepted_on_complete_;
420 grpc_closure on_complete_ =
421 MakeMemberClosure<SendMessage, &SendMessage::OnComplete>(this);
422 absl::Status completed_status_;
423 };

425 // State machine for receiving messages: handles intercepting recv_message
426 // ops, forwarding them down the stack, and then publishing the result via
427 // pipes to the promise (and ultimately calling the right callbacks for the
428 // batch when our promise has completed processing of them).
429 // Split into its own class so that we don't spend the memory instantiating
430 // these members for filters that don't need to intercept sent messages.
431 class ReceiveMessage {
432 public:
433 ReceiveMessage(BaseCallData* base, Interceptor* interceptor)
434 : base_(base), interceptor_(interceptor) {}
435 ~ReceiveMessage() { interceptor_->~Interceptor(); }
436
437 Interceptor* interceptor() { return interceptor_; }
438
439 // Start a recv_message op.
440 void StartOp(CapturedBatch& batch);
441 // Publish the inbound pipe to the filter.
442 // This happens when the promise requests to call the next filter: until
443 // this occurs messages can't be received as we don't know the pipe that the
444 // promise expects to forward them with.
445 template <typename T>
446 void GotPipe(T* pipe);
447 // Called from client/server polling to do the receive message part of the
448 // work.
449 void WakeInsideCombiner(Flusher* flusher, bool allow_push_to_pipe);
450 // Call is completed, we have trailing metadata. Close things out.
451 void Done(const ServerMetadata& metadata, Flusher* flusher);
452
453 private:
454 enum class State : uint8_t {
455 // Starting state: no batch started, no incoming pipe configured.
456 kInitial,
457 // We have an incoming pipe, but no batch started.
458 // (this is the steady state).
459 kIdle,
460 // We received a batch and forwarded it on, but have not got an incoming
461 // pipe configured.
462 kForwardedBatchNoPipe,
463 // We received a batch and forwarded it on.
464 kForwardedBatch,
465 // We got the completion for the recv_message, but we don't yet have a
466 // pipe configured. Stall until this changes.
467 kBatchCompletedNoPipe,
468 // We got the completion for the recv_message, and we have a pipe
469 // configured: next poll will push the message into the pipe for the
470 // filter to process.
471 kBatchCompleted,
472 // We've pushed a message into the promise, and we're now waiting for it
473 // to pop out the other end so we can forward it up the stack.
474 kPushedToPipe,
475 // We've got a message out of the pipe, now we need to wait for processing
476 // to completely quiesce in the promise prior to forwarding the completion
477 // up the stack.
478 kPulledFromPipe,
479 // We're done.
480 kCancelled,
481 // Call got terminated whilst we were idle: we need to close the sender
482 // pipe next poll.
483 kCancelledWhilstIdle,
484 // Call got terminated whilst we had forwarded a recv_message down the
485 // stack: we need to keep track of that until we get the completion so
486 // that we do the right thing in OnComplete.
487 kCancelledWhilstForwarding,
488 // The same, but before we got the pipe
489 kCancelledWhilstForwardingNoPipe,
490 // Call got terminated whilst we had a recv_message batch completed, and
491 // we've now received the completion.
492 // On the next poll we'll close things out and forward on completions,
493 // then transition to cancelled.
494 kBatchCompletedButCancelled,
495 // The same, but before we got the pipe
496 kBatchCompletedButCancelledNoPipe,
497 // Completed successfully while we're processing a recv message - see
498 // kPushedToPipe.
499 kCompletedWhilePushedToPipe,
500 // Completed successfully while we're processing a recv message - see
501 // kPulledFromPipe.
502 kCompletedWhilePulledFromPipe,
503 // Completed successfully while we were waiting to process
504 // kBatchCompleted.
505 kCompletedWhileBatchCompleted,
506 };
507 static const char* StateString(State);
508
509 void OnComplete(absl::Status status);
510
511 BaseCallData* const base_;
512 Interceptor* const interceptor_;
513 State state_ = State::kInitial;
514 uint32_t scratch_flags_;
515 absl::optional<SliceBuffer>* intercepted_slice_buffer_;
516 uint32_t* intercepted_flags_;
517 absl::optional<PipeSender<MessageHandle>::PushType> push_;
518 absl::optional<PipeReceiverNextType<MessageHandle>> next_;
519 absl::Status completed_status_;
520 grpc_closure* intercepted_on_complete_;
521 grpc_closure on_complete_ =
522 MakeMemberClosure<ReceiveMessage, &ReceiveMessage::OnComplete>(this);
523 };

596 // Start state: no op seen

606 kComplete,
607 // We've called the recv_metadata_ready callback from the original
608 // recv_trailing_metadata op that was presented to us.
609 kResponded,
610 // We've been cancelled and handled that locally.
611 // (i.e. whilst the recv_trailing_metadata op is queued in this filter).
612 kCancelled
613 };
