49class TraceFlag {
50 public:
51 TraceFlag(bool default_enabled, const char* name);
52 // TraceFlag needs to be trivially destructible since it is used as global
53 // variable.
54 ~TraceFlag() = default;
55
56 const char* name() const { return name_; }
57
58// Use the symbol GRPC_USE_TRACERS to determine if tracers will be enabled in
59// opt builds (tracers are always on in dbg builds). The default in OSS is for
60// tracers to be on since we support binary distributions of gRPC for the
61// wrapped language (wr don't want to force recompilation to get tracing).
62// Internally, however, for performance reasons, we compile them out by
63// default, since internal build systems make recompiling trivial.
64//
65// Prefer GRPC_TRACE_FLAG_ENABLED() macro instead of using enabled() directly.
66#define GRPC_USE_TRACERS // tracers on by default in OSS
67#if defined(GRPC_USE_TRACERS) || !defined(NDEBUG)
68 bool enabled() { return value_.load(std::memory_order_relaxed); }
69#else
70 bool enabled() { return false; }
71#endif // defined(GRPC_USE_TRACERS) || !defined(NDEBUG)
72
73 private:
74 friend void testing::grpc_tracer_enable_flag(TraceFlag* flag);
75 friend class TraceFlagList;
76
77 void set_enabled(bool enabled) {
78 value_.store(enabled, std::memory_order_relaxed);
79 }
80
81 TraceFlag* next_tracer_;
82 const char* const name_;
83 std::atomic<bool> value_;
84};
