45class XdsHttpFilterImpl {
46 public:
47 struct FilterConfig {
48 absl::string_view config_proto_type_name;
49 Json config;
50
51 bool operator==(const FilterConfig& other) const {
52 return config_proto_type_name == other.config_proto_type_name &&
53 config == other.config;
54 }
55 std::string ToString() const {
56 return absl::StrCat("{config_proto_type_name=", config_proto_type_name,
57 " config=", JsonDump(config), "}");
58 }
59 };
60
61 // Service config data for the filter, returned by GenerateServiceConfig().
62 struct ServiceConfigJsonEntry {
63 // The top-level field name in the method config.
64 // Filter implementations should use their primary config proto type
65 // name for this.
66 // The value of this field in the method config will be a JSON array,
67 // which will be populated with the elements returned by each filter
68 // instance.
69 std::string service_config_field_name;
70 // The element to add to the JSON array.
71 std::string element;
72 };
73
74 virtual ~XdsHttpFilterImpl() = default;
75
76 // Returns the top-level filter config proto message name.
77 virtual absl::string_view ConfigProtoName() const = 0;
78
79 // Returns the override filter config proto message name.
80 // If empty, no override type is supported.
81 virtual absl::string_view OverrideConfigProtoName() const = 0;
82
83 // Loads the proto message into the upb symtab.
84 virtual void PopulateSymtab(upb_DefPool* symtab) const = 0;
85
86 // Generates a Config from the xDS filter config proto.
87 // Used for the top-level config in the HCM HTTP filter list.
88 virtual absl::optional<FilterConfig> GenerateFilterConfig(
89 const XdsResourceType::DecodeContext& context, XdsExtension extension,
90 ValidationErrors* errors) const = 0;
91
92 // Generates a Config from the xDS filter config proto.
93 // Used for the typed_per_filter_config override in VirtualHost and Route.
94 virtual absl::optional<FilterConfig> GenerateFilterConfigOverride(
95 const XdsResourceType::DecodeContext& context, XdsExtension extension,
96 ValidationErrors* errors) const = 0;
97
98 // C-core channel filter implementation.
99 virtual const grpc_channel_filter* channel_filter() const = 0;
100
101 // Modifies channel args that may affect service config parsing (not
102 // visible to the channel as a whole).
103 virtual ChannelArgs ModifyChannelArgs(const ChannelArgs& args) const {
104 return args;
105 }
106
107 // Function to convert the Configs into a JSON string to be added to the
108 // per-method part of the service config.
109 // The hcm_filter_config comes from the HttpConnectionManager config.
110 // The filter_config_override comes from the first of the ClusterWeight,
111 // Route, or VirtualHost entries that it is found in, or null if
112 // there is no override in any of those locations.
113 virtual absl::StatusOr<ServiceConfigJsonEntry> GenerateServiceConfig(
114 const FilterConfig& hcm_filter_config,
115 const FilterConfig* filter_config_override,
116 absl::string_view filter_name) const = 0;
117
118 // Returns true if the filter is supported on clients; false otherwise
119 virtual bool IsSupportedOnClients() const = 0;
120
121 // Returns true if the filter is supported on servers; false otherwise
122 virtual bool IsSupportedOnServers() const = 0;
123
124 // Returns true if the filter must be the last filter in the chain.
125 virtual bool IsTerminalFilter() const { return false; }
126};

128class XdsHttpRouterFilter : public XdsHttpFilterImpl {
129 public:
130 absl::string_view ConfigProtoName() const override;
131 absl::string_view OverrideConfigProtoName() const override;
132 void PopulateSymtab(upb_DefPool* symtab) const override;
133 absl::optional<FilterConfig> GenerateFilterConfig(

135 ValidationErrors* errors) const override;
136 absl::optional<FilterConfig> GenerateFilterConfigOverride(
137 const XdsResourceType::DecodeContext& context, XdsExtension extension,
138 ValidationErrors* errors) const override;
139 const grpc_channel_filter* channel_filter() const override { return nullptr; }
140 absl::StatusOr<ServiceConfigJsonEntry> GenerateServiceConfig(
141 const FilterConfig& /*hcm_filter_config*/,
142 const FilterConfig* /*filter_config_override*/,
143 absl::string_view /*filter_name*/) const override {
144 // This will never be called, since channel_filter() returns null.
145 return absl::UnimplementedError("router filter should never be called");
146 }
147 bool IsSupportedOnClients() const override { return true; }
148 bool IsSupportedOnServers() const override { return true; }
149 bool IsTerminalFilter() const override { return true; }
150};
